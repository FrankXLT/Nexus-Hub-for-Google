        Container(webapp, "Frontend Orchestrator", "HTML/JS/GAS", "Delivers UI, Zero-Trust visual flow, configures rules.")

        Container(worker, "Sync Engine / Workers", "Python", "Runs delta syncs, extracts data, calls LLMs.")
        ContainerDb(db, "Nexus SQLite DB", "SQLite WAL", "Stores taxonomy, artifacts, config, history, queues.")
    }
    
    System_Ext(google_ws, "Google Workspace API", "Gmail, Drive, Contacts API")
    System_Ext(gemini, "Google Gemini API", "LLM Inference")

    Rel(user, webapp, "Views telemetry, sets rules", "HTTPS")
    Rel(webapp, api, "Fetches stats, saves config", "HTTPS/HMAC")
    Rel(api, worker, "Dispatches background tasks", "In-Process")
    Rel(api, db, "Reads/Writes State", "SQL")
    Rel(worker, db, "Reads/Writes State", "SQL")
    Rel(worker, google_ws, "Extracts Artifacts", "OAuth2")
    Rel(worker, gemini, "Classifies & Profiles", "REST")
